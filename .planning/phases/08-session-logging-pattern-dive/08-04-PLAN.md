import { getSessionsForProject, getProjectSessionStats, getActiveProjectsForPicker } from "@/lib/actions/session-actions";

export default async function ChartDetailPage({ params }: { params: Promise<{ id: string }> }) {
  const { id } = await params;
  const chart = await getChart(id);
  if (!chart) notFound();

  const [projectSupplies, imageUrls, sessions, sessionStats, activeProjects] = await Promise.all([
    chart.project ? getProjectSupplies(chart.project.id) : null,
    getPresignedImageUrls([chart.coverImageUrl, chart.coverThumbnailUrl]),
    chart.project ? getSessionsForProject(chart.project.id) : [],
    chart.project ? getProjectSessionStats(chart.project.id) : { totalStitches: 0, sessionsLogged: 0, avgPerSession: 0, activeSince: null },
    getActiveProjectsForPicker(),
  ]);

  // Also resolve session photo keys to presigned URLs
  const sessionPhotoKeys = sessions.filter(s => s.photoKey).map(s => s.photoKey);
  const allImageUrls = sessionPhotoKeys.length > 0
    ? { ...imageUrls, ...(await getPresignedImageUrls(sessionPhotoKeys)) }
    : imageUrls;

  return (
    <ProjectDetailPage
      chart={chart}
      imageUrls={allImageUrls}
      supplies={projectSupplies}
      sessions={sessions}
      sessionStats={sessionStats}
      activeProjects={activeProjects}
    />
  );
}
