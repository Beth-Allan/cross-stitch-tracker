```typescript
"use server";
import { revalidatePath } from "next/cache";
import { z } from "zod";
import { requireAuth } from "@/lib/auth-guard";
import { prisma } from "@/lib/db";

export async function createThread(formData: unknown) {
  await requireAuth();
  try {
    const validated = threadSchema.parse(formData);
    const thread = await prisma.thread.create({ data: validated });
    revalidatePath("/supplies");
    return { success: true as const, thread };
  } catch (error) {
    if (error instanceof z.ZodError) {
      return { success: false as const, error: error.errors[0].message };
    }
    return { success: false as const, error: "Failed to create thread" };
  }
}
```

<feature>
  <name>Session CRUD with Atomic Progress Recalculation</name>
  <files>src/lib/actions/session-actions.ts, src/lib/actions/session-actions.test.ts</files>
  <behavior>
    **createSession:**
    - Requires auth (rejects unauthenticated)
    - Validates input with sessionFormSchema (rejects invalid)
    - Verifies project.userId === user.id (rejects unauthorized)
    - Creates StitchSession inside $transaction
    - Recalculates stitchesCompleted = startingStitches + sum(session.stitchCount) atomically
    - Calls revalidatePath for `/charts/{chartId}` and `/sessions`
    - Returns { success: true, session } on success

    **updateSession:**
    - Requires auth
    - Finds existing session, verifies project ownership
    - Updates session fields inside $transaction
    - Recalculates progress atomically
    - Returns { success: true, session }

    **deleteSession:**
    - Requires auth
    - Finds existing session, verifies project ownership
    - Deletes session inside $transaction
    - Recalculates progress atomically (sum decreases)
    - Returns { success: true }

    **getSessionsForProject(projectId):**
    - Requires auth, verifies project ownership
    - Returns sessions ordered by date desc with StitchSessionRow shape

    **getAllSessions():**
    - Requires auth
    - Returns all user's sessions across projects, ordered by date desc
    - Includes project name (via chart.name)

    **getActiveProjectsForPicker():**
    - Requires auth
    - Returns projects with status IN_PROGRESS, ON_HOLD, KITTING, or KITTED
    - Returns ActiveProjectForPicker shape with chartName, coverThumbnailUrl, totalStitches

    **getProjectSessionStats(projectId):**
    - Requires auth, verifies project ownership
    - Returns ProjectSessionStats: totalStitches, sessionsLogged, avgPerSession, activeSince
  </behavior>
  <implementation>
Create `src/lib/actions/session-actions.ts` following the established server action pattern (supply-actions.ts for CRUD, chart-actions.ts for $transaction and ownership checks).

Key implementation details:
- `recalculateProgress(tx, projectId)` helper: `tx.stitchSession.aggregate({ where: { projectId }, _sum: { stitchCount: true } })` then `tx.project.update({ data: { stitchesCompleted: startingStitches + (sum ?? 0) } })`. Must fetch `startingStitches` inside the transaction per D-04.
- Active project statuses for picker: `["IN_PROGRESS", "ON_HOLD", "KITTING", "KITTED"]` per RESEARCH.md Open Question 1 recommendation.
- Session queries join through project to get userId for ownership filtering: `where: { project: { userId: user.id } }`.
- getAllSessions includes chartName via: `include: { project: { include: { chart: { select: { id: true, name: true } } } } }`.
- For updateSession, the sessionId is the first param, formData is the second. Ownership is verified by finding the session's project and checking userId.
  </implementation>
