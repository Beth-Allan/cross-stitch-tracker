    **Create src/lib/r2.ts** -- S3Client singleton for Cloudflare R2 (per D-05, D-08):
    ```typescript
    import { S3Client } from "@aws-sdk/client-s3";

    function createR2Client(): S3Client {
      if (!process.env.R2_ACCOUNT_ID || !process.env.R2_ACCESS_KEY_ID || !process.env.R2_SECRET_ACCESS_KEY) {
        throw new Error(
          "R2 environment variables not configured. Set R2_ACCOUNT_ID, R2_ACCESS_KEY_ID, and R2_SECRET_ACCESS_KEY."
        );
      }
      return new S3Client({
        region: "auto",
        endpoint: `https://${process.env.R2_ACCOUNT_ID}.r2.cloudflarestorage.com`,
        credentials: {
          accessKeyId: process.env.R2_ACCESS_KEY_ID,
          secretAccessKey: process.env.R2_SECRET_ACCESS_KEY,
        },
      });
    }

    // Lazy singleton -- only created when first accessed (avoids errors when R2 not configured)
    let _r2Client: S3Client | null = null;
    export function getR2Client(): S3Client {
      if (!_r2Client) _r2Client = createR2Client();
      return _r2Client;
    }

    export const R2_BUCKET_NAME = process.env.R2_BUCKET_NAME ?? "cross-stitch-tracker";
    ```
    Note: Use lazy singleton so the app doesn't crash on import when R2 env vars are missing (graceful degradation per Research open question 1).

    **Create src/lib/utils/size-category.ts** -- per D-04, computed at query time:
    ```typescript
    export type SizeCategory = "Mini" | "Small" | "Medium" | "Large" | "BAP";

    export function calculateSizeCategory(stitchCount: number): SizeCategory {
      if (stitchCount >= 50_000) return "BAP";
      if (stitchCount >= 25_000) return "Large";
      if (stitchCount >= 5_000) return "Medium";
      if (stitchCount >= 1_000) return "Small";
      return "Mini";
    }

    export function getEffectiveStitchCount(
      stitchCount: number,
      stitchesWide: number,
      stitchesHigh: number,
    ): { count: number; approximate: boolean } {
      if (stitchCount > 0) return { count: stitchCount, approximate: false };
      if (stitchesWide > 0 && stitchesHigh > 0) return { count: stitchesWide * stitchesHigh, approximate: true };
      return { count: 0, approximate: false };
    }
    ```
