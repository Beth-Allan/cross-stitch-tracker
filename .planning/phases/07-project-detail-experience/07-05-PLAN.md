    const [highlightIndex, setHighlightIndex] = useState(-1);

    function handleKeyDown(e: React.KeyboardEvent) {
      if (e.key === "ArrowDown") {

        setHighlightIndex(prev => Math.min(prev + 1, items.length - 1));

        e.preventDefault();
        setHighlightIndex(prev => Math.max(prev - 1, 0));
      }
      // ... Enter handling
    }
